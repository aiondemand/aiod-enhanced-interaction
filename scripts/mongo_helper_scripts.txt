# run this command within MongoDB container to export a collection as a JSON file
mongoexport --db=<DB_NAME> --collection=<COL_NAME> --out=cols.json --jsonArray --username=<MONGO_USER> --password=<MONGO_PASS> --authenticationDatabase=admin

# mongoexport --db=aiod --collection=assetCollections --out=cols.json --jsonArray --username=aiod --password=mongopassword --authenticationDatabase=admin

# run this command within MongoDB container to import a collection from a JSON file
mongoimport --db=<DB_NAME> --collection=<COL_NAME> --file=cols.json --jsonArray --username=<MONGO_USER> --password=<MONGO_PASS> --authenticationDatabase=admin
